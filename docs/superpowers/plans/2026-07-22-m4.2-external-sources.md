def migrate_m42_document_fields() -> None:
    """M4.2 迁移：为 Document 表新增 6 个字段（若不存在）。

    SQLite 不支持 ALTER TABLE ADD COLUMN IF NOT EXISTS，需检查后添加。
    向后兼容：新字段均有默认值，旧数据自动获得默认值。
    """
    import sqlite3

    db_path = get_db_path()
    conn = sqlite3.connect(str(db_path))
    cursor = conn.cursor()
    # 获取现有列
    cursor.execute("PRAGMA table_info(document)")
    existing_cols = {row[1] for row in cursor.fetchall()}
    # 新增列定义
    new_cols = [
        ("source", "TEXT DEFAULT 'local'"),
        ("source_id", "TEXT"),
        ("verified", "BOOLEAN DEFAULT 1"),
        ("season", "TEXT"),
        ("hidden", "BOOLEAN DEFAULT 0"),
        ("status", "TEXT DEFAULT 'published'"),
    ]
    for col_name, col_def in new_cols:
        if col_name not in existing_cols:
            cursor.execute(f"ALTER TABLE document ADD COLUMN {col_name} {col_def}")
    conn.commit()
    conn.close()
```

def normalize_ingredient(en_name: str) -> str | None:
    """英文材料名 → 中文标准名。

    先查 _INGREDIENT_OVERRIDES，再查 ingredients.canonicalize。
    返回 None 表示无法归一化。
    """
    if not en_name:
        return None
    key = en_name.strip().lower()
    if key in _INGREDIENT_OVERRIDES:
        return _INGREDIENT_OVERRIDES[key]
    # 尝试 canonicalize（可能命中已注册的中英文别名）
    result = canonicalize(en_name)
    if result != en_name.lower():
        return result
    return None


def parse_recipe(api_data: dict[str, Any]) -> dict[str, Any]:
    """解析 TheCocktailDB 单条 API 响应为配方 dict。"""
    source_id = api_data.get("idDrink", "")
    title = api_data.get("strDrink", "")
    instructions = api_data.get("strInstructions", "")

    # 解析材料（strIngredient1-15 + strMeasure1-15）
    ingredients: list[str] = []
    measures: list[str] = []
    unknown: list[str] = []
    for i in range(1, 16):
        ing = api_data.get(f"strIngredient{i}")
        measure = api_data.get(f"strMeasure{i}", "")
        if not ing:
            break
        normalized = normalize_ingredient(ing)
        if normalized:
            ingredients.append(normalized)
            measures.append(measure.strip())
        else:
            unknown.append(ing)

    # 构建内容 Markdown
    content_lines = [f"# {title}\n\n## 配方"]
    for ing, measure in zip(ingredients, measures):
        content_lines.append(f"- {ing} {measure}".rstrip())
    content_lines.append(f"\n## 步骤\n{instructions}")
    if unknown:
        content_lines.append(f"\n## 未归一化材料\n{', '.join(unknown)}")
    content = "\n".join(content_lines)

    return {
        "title": title,
        "source_id": source_id,
        "ingredients": ingredients,
        "content": content,
        "source": "thecocktaildb",
        "verified": False,
        "unknown_ingredients": unknown,
    }


def sync_thecocktaildb(limit: int = 50) -> dict[str, Any]:
    """从 TheCocktailDB 同步配方。

    Args:
        limit: 最多拉取条数

    Returns:
        {imported, skipped, failed, unknown_ingredients}
    """
    imported = 0
    skipped = 0
    failed = 0
    all_unknown: list[str] = []

    # 拉取配方列表
    url = f"{API_BASE}/search.php?f=a"  # 按首字母搜索，返回以 a 开头的配方
    try:
        resp = httpx.get(url, timeout=30)
        resp.raise_for_status()
        data = resp.json()
    except Exception:
        return {"imported": 0, "skipped": 0, "failed": 1, "unknown_ingredients": []}

    drinks = data.get("drinks") or []
    importer = ImportService()

    for drink in drinks[:limit]:
        try:
            recipe = parse_recipe(drink)
            all_unknown.extend(recipe.pop("unknown_ingredients", []))

            # 去重：source + source_id
            with get_session() as session:
                existing = session.exec(
                    select(Document).where(
                        Document.source == "thecocktaildb",
                        Document.source_id == recipe["source_id"],
                    )
                ).first()
                if existing:
                    skipped += 1
                    continue

            # 导入
            result = importer.import_text(
                content=recipe["content"],
                title=recipe["title"],
                source_type="thecocktaildb",
                file_type="md",
            )
            # 设置 category + 数据源字段
            doc_id = result.get("doc_id")
            if doc_id:
                with get_session() as session:
                    doc = session.get(Document, doc_id)
                    if doc:
                        doc.category = "recipe"
                        doc.source = "thecocktaildb"
                        doc.source_id = recipe["source_id"]
                        doc.verified = False
                        session.add(doc)
                        session.commit()
                imported += 1
            else:
                failed += 1
        except Exception:
            failed += 1

    return {
        "imported": imported,
        "skipped": skipped,
        "failed": failed,
        "unknown_ingredients": list(set(all_unknown)),
    }
```

```python
"""配方筛选/审核/隐藏（M4.2 数据源治理）。"""
from __future__ import annotations

from typing import Any

from sqlmodel import select

from hermes_kb.database import get_session
from hermes_kb.models import Document


def filter_recipes(
    source: str | None = None,
    verified: bool | None = None,
    hidden: bool | None = None,
    status: str | None = None,
    limit: int = 100,
) -> list[dict[str, Any]]:
    """筛选配方列表。

    Args:
        source: 数据源筛选（thecocktaildb/ima/iba/user/ugc/local）
        verified: 审核状态筛选
        hidden: 隐藏状态筛选
        status: 状态筛选（draft/pending/published/rejected）
        limit: 最多返回条数

    Returns:
        [{doc_id, title, source, source_id, verified, season, hidden, status}]
    """
    with get_session() as session:
        stmt = select(Document).where(Document.category == "recipe")
        if source is not None:
            stmt = stmt.where(Document.source == source)
        if verified is not None:
            stmt = stmt.where(Document.verified == verified)
        if hidden is not None:
            stmt = stmt.where(Document.hidden == hidden)
        if status is not None:
            stmt = stmt.where(Document.status == status)
        stmt = stmt.limit(limit)
        docs = session.exec(stmt).all()
        return [
            {
                "doc_id": d.doc_id,
                "title": d.title,
                "source": d.source,
                "source_id": d.source_id,
                "verified": d.verified,
                "season": d.season,
                "hidden": d.hidden,
                "status": d.status,
            }
            for d in docs
        ]


def verify_recipe(doc_id: str) -> bool:
    """审核通过配方（verified=True, status=published）。"""
    with get_session() as session:
        doc = session.get(Document, doc_id)
        if not doc:
            return False
        doc.verified = True
        doc.status = "published"
        session.add(doc)
        session.commit()
        return True


def hide_recipe(doc_id: str, hidden: bool = True) -> bool:
    """隐藏/取消隐藏配方。"""
    with get_session() as session:
        doc = session.get(Document, doc_id)
        if not doc:
            return False
        doc.hidden = hidden
        session.add(doc)
        session.commit()
        return True
```

    @app.post("/api/lab/sync")
    async def lab_sync(req: dict[str, Any]) -> dict[str, Any]:
        """触发外部数据源同步。"""
        from hermes_kb.thecocktaildb_sync import sync_thecocktaildb
        from hermes_kb.ima_adapter import fetch_ima_recipes

        source = (req.get("source") or "thecocktaildb").strip()
        limit = min(int(req.get("limit") or 50), 200)

        if source == "thecocktaildb":
            return sync_thecocktaildb(limit=limit)
        elif source == "ima":
            api_key = req.get("api_key") or ""
            return fetch_ima_recipes(api_key=api_key, limit=limit)
        else:
            raise HTTPException(status_code=400, detail=f"不支持的 source: {source}")

    @app.get("/api/lab/recipes")
    async def lab_recipes(
        source: str | None = None,
        verified: bool | None = None,
        hidden: bool | None = None,
        status: str | None = None,
        limit: int = 100,
    ) -> dict[str, Any]:
        """按来源/状态筛选配方。"""
        from hermes_kb.recipe_filter import filter_recipes

        limit = max(1, min(limit, 500))
        items = filter_recipes(
            source=source, verified=verified, hidden=hidden, status=status, limit=limit
        )
        return {"items": items, "count": len(items)}

    @app.post("/api/lab/recipes/{doc_id}/verify")

        """审核通过配方。"""
        from hermes_kb.recipe_filter import verify_recipe

        ok = verify_recipe(doc_id)
        if not ok:
            raise HTTPException(status_code=404, detail="配方不存在")
        return {"doc_id": doc_id, "status": "ok"}

        if (d.unknown_ingredients && d.unknown_ingredients.length > 0) {
